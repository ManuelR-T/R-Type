120 bool removeClient(size_t packetId, UDPServer::client_id_t clientId)
121 {
122 std::lock_guard<std::mutex> lck(_mut);
123 for (auto &timeoutPacket : _timeoutPackets) {
124 if (packetId == timeoutPacket.packetId) {
125 auto itCli = std::find(timeoutPacket.clientIds.begin(), timeoutPacket.clientIds.end(), clientId);
126 if (itCli == timeoutPacket.clientIds.end()) {
127 eng::logInfo(std::format("No such client ({}) in packet ({})", clientId, packetId));
128 return false;
129 }
130 timeoutPacket.clientIds.erase(itCli);
131 if (timeoutPacket.clientIds.empty()) {
