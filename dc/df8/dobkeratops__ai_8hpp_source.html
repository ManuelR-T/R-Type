136void cleanupDobkeratopsSegments(ecs::Registry &reg, entity_t bossEntity, std::list<std::vector<char>> &datasToSend)
137{
138 if (!dobkeratopsStates.contains(bossEntity)) {
139 eng::logWarning("Attempted to cleanup non-existent Dobkeratops state");
140 return;
141 }
142
143 auto &state = dobkeratopsStates[bossEntity];
144
145 std::vector<entity_t> segmentsToRemove = state.neckSegments;
146
147 state.neckSegments.clear();
148
149 for (auto segment : segmentsToRemove) {

162 reg.killEntity(segment);
163
164 } catch (const std::exception &e) {
165 eng::logError("Failed to cleanup segment " + std::to_string(segment) + ": " + e.what());
166 }
167 }
168 dobkeratopsStates.erase(bossEntity);
169 try {
170 auto sharedId = reg.getComponent<ecs::component::SharedEntity>(state.bossParasite)->sharedEntityId;
171 datasToSend.push_back(
172 rt::UDPPacket<rt::UDPBody::DEL_ENTITY>(rt::UDPCommand::DEL_ENTITY, sharedId, true).serialize()
173 );
174 reg.killEntity(state.bossParasite);
175 } catch (const std::exception &e) {
176 eng::logError("Failed to cleanup boss parasite: " + std::string(e.what()));
177 }

180void dobkeratopsAi(ecs::Registry &reg, entity_t e, std::list<std::vector<char>> &datasToSend)
181{
182 auto &state = dobkeratopsStates[e];
183 auto bossPos = reg.getComponent<ecs::component::Position>(e);
184 auto vel = reg.getComponent<ecs::component::Velocity>(e);
185 auto health = reg.getComponent<ecs::component::Health>(state.bossParasite);
186 auto anim = reg.getComponent<ecs::component::Animation>(e);
187 auto sharedEntity = reg.getComponent<ecs::component::SharedEntity>(e);
188
189 if (!bossPos || !vel || !health || !anim || !sharedEntity) {
190 eng::logWarning("Missing required components for Dobkeratops AI");
191 return;
192 }
193 state.attackTicks++;
194 state.neckTicks++;
195
196 const float maxSegmentDistance = 20.0f;
197 const float segmentLength = 16.0f;
198 const float initialXOffset = 58.0f;
199 const float initialYOffset = 180.0f;
200
201 for (size_t i = 0; i < state.neckSegments.size(); i++) {
202 entity_t segmentEntity = state.neckSegments[i];
203 if (!reg.hasComponent<ecs::component::Position>(segmentEntity)) {
204 continue;
205 }
206
207 auto &segPos = reg.getComponent<ecs::component::Position>(segmentEntity);
208
209 if (i == 0) {
210 float targetX = bossPos->x + initialXOffset;
211 float targetY = bossPos->y + initialYOffset;
212
213 segPos->x = targetX;
214 segPos->y = targetY;
215 } else {
216 auto prevSegPos = reg.getComponent<ecs::component::Position>(state.neckSegments[i - 1]);
217 if (!prevSegPos) {
218 continue;
219 }
220
221 float targetX = prevSegPos->x - segmentLength;
222 float baseY = prevSegPos->y - (segmentLength * 0.2f);
223
224 float amplitude = std::min(6.0f * i, 25.0f);
225 float delay = i * 0.3f;
226 float targetY = baseY + std::sin(state.neckTicks * 0.05f - delay) * amplitude;
227
228 float dx = targetX - prevSegPos->x;
229 float dy = targetY - prevSegPos->y;
230 float distance = std::sqrt(dx * dx + dy * dy);
231 if (distance > maxSegmentDistance) {
232 float ratio = maxSegmentDistance / distance;
233 targetX = prevSegPos->x + dx * ratio;
234 targetY = prevSegPos->y + dy * ratio;
235 }
236
237 segPos->x = targetX;
238 segPos->y = targetY;
239 }
240 }
241
242 if (state.attackTicks >= 90) {
243 state.attackTicks = 0;
244 state.isAttacking = true;
245 state.projectilesShot = 0;
246 state.attackPattern = (state.attackPattern + 1) % 3;
247 anim->state = "attack";
248 sendAnimationStateChange(datasToSend, sharedEntity->sharedEntityId, "attack");
249 }
250
251 if (state.isAttacking) {
252 switch (state.attackPattern) {
253 case 0:
254 if (state.projectilesShot < 1) {
255 float angle = eng::RandomGenerator::generate(-0.5f, 0.5f);
256 spawnDobkeratopsProjectile(reg, e, datasToSend, angle);
257 state.projectilesShot++;
258 }
259 break;
260
261 case 1:
262 if (state.projectilesShot < 8) {
263 float baseAngle = -0.5f;
264 float angleStep = 0.2f;
265 spawnDobkeratopsProjectile(reg, e, datasToSend, baseAngle + angleStep * state.projectilesShot);
266 state.projectilesShot++;
267 }
268 break;
269
270 case 2:
271 if (state.projectilesShot < 20 && (state.attackTicks % 3 == 0)) {
272 float angle = eng::RandomGenerator::generate(-0.7f, 0.7f);
273 spawnDobkeratopsProjectile(reg, e, datasToSend, angle);
274 state.projectilesShot++;
275 }
276 break;
277 }
278
279 if ((state.attackPattern != 2 && state.projectilesShot >= 8) ||
280 (state.attackPattern == 2 && state.projectilesShot >= 20)) {
281 state.isAttacking = false;
282 anim->state = "idle";
283 sendAnimationStateChange(datasToSend, sharedEntity->sharedEntityId, "idle");
284 }
285 }
286
287 if (state.phase == 0 && health->currHp <= 75) {
288 auto shared = reg.getComponent<ecs::component::SharedEntity>(state.bossParasite);
289 sendAnimationStateChange(datasToSend, shared->sharedEntityId, "p1");
290 state.phase = 1;
291 state.attackTicks = -20;
292 }
293
294 if (state.phase == 1 && state.attackTicks >= 0) {
295 auto shared = reg.getComponent<ecs::component::SharedEntity>(state.bossParasite);
296 sendAnimationStateChange(datasToSend, shared->sharedEntityId, "p2");
297 state.phase = 2;
298 }
299
300 if (health->currHp <= 1) {
301 cleanupDobkeratopsSegments(reg, e, datasToSend);
302 datasToSend.push_back(
303 rt::UDPPacket<rt::UDPBody::DEL_ENTITY>(rt::UDPCommand::DEL_ENTITY, sharedEntity->sharedEntityId, true)
304 .serialize()
305 );
306 reg.killEntity(e);
307 return;
308 }
309}
