176inline bool ArgParser::getValue<bool>(const std::string &longName) const
177{
178 auto it = std::find_if(_arguments.begin(), _arguments.end(), [&](const Argument &arg) {
179 return arg.long_name == longName;
180 });
181
182 if (it != _arguments.end() && it->value) {
183 std::string val = *(it->value);
184 std::transform(val.begin(), val.end(), val.begin(), ::tolower);
185 if (val == "true" || val == "1") {
186 return true;
187 } else if (val == "false" || val == "0") {
188 return false;
