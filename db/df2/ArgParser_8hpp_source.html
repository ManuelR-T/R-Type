65template <typename T>
66T ArgParser::getValue(const std::string &longName) const
67{
68 auto it = std::find_if(_arguments.begin(), _arguments.end(), [&](const Argument &arg) {
69 return arg.long_name == longName;
70 });
71
72 if (it != _arguments.end() && it->value) {
73 T result;
74 std::istringstream iss(*(it->value));
75 iss >> result;
76 if (iss.fail() || !iss.eof()) {
77 throw std::runtime_error(
78 "Failed to convert argument value to the requested type for argument: " + longName
79 );
80 }
81 return result;
82 }
83 throw std::runtime_error("Argument is not defined or has no value: " + longName);
84}

86template <>
87inline bool ArgParser::getValue<bool>(const std::string &longName) const
88{
89 auto it = std::find_if(_arguments.begin(), _arguments.end(), [&](const Argument &arg) {
90 return arg.long_name == longName;
91 });
92
93 if (it != _arguments.end() && it->value) {
94 std::string val = *(it->value);
95 std::transform(val.begin(), val.end(), val.begin(), ::tolower);
96 if (val == "true" || val == "1") {
97 return true;
98 } else if (val == "false" || val == "0") {
99 return false;
100 } else {
101 throw std::runtime_error("Invalid boolean value for argument: " + longName);
102 }
103 }
104 throw std::runtime_error("Argument is not defined or has no value: " + longName);
105}
